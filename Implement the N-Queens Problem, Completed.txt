def dfs_n_queens(n):
    if n < 1:
        return []

    solutions = []

    def backtrack(row, current_solution):
        if row == n:
            solutions.append(current_solution[:])
            return

        for col in range(n):
            if is_safe(row, col, current_solution):
                current_solution.append(col)
                backtrack(row + 1, current_solution)
                current_solution.pop()

    def is_safe(row, col, current_solution):
        for prev_row in range(len(current_solution)):
            prev_col = current_solution[prev_row]

            if col == prev_col:
                return False

            if abs(row - prev_row) == abs(col - prev_col):
                return False

        return True

    backtrack(0, [])
    return solutions
